void bfs(int adj[MAX][MAX], int visited[MAX],int v,int start){

    int queue[v], rear=-1,front=-1,i,k;
    for(k = 0 ;k < v; k++){
        visited[k]=0;
    }
    queue[++rear]=start;
    ++front;
    visited[start]=1;
    while(rear>=front){
        start = queue[front++];
        printf("%d->",start);
        for(i = 0 ; i < v ; i++){
            if(adj[start][i]&&visited[i]==0){
                queue[++rear]=i;
                visited[i]=1;
            }
        }
    }
}
